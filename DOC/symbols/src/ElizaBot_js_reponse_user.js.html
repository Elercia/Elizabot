  2 /**
  3  * Fonction permettant de donner une réponse à l'utilisateur en fonction des mots clés récupérés
  4  *
  5  * @method     donner_reponse
  6  */
  7 ElizaBot.prototype.donner_reponse = function()
  8 {
  9 	//récupere la saisie de l'utilisateur
 10 	var saisie_utilisateur = document.getElementById("user_input_text").value;
 11 	saisie_utilisateur=saisie_utilisateur.trim();
 12 	//Si la saisie ne ressemble pas à une commande
 13 	if (saisie_utilisateur[0] != "/")
 14 	{
 15 		//S'il y a bien quelque chose d'écrit
 16 		if (saisie_utilisateur != "")
 17 		{
 18 			this.derniers_messages.push(saisie_utilisateur);
 19 
 20 			//recherche le ou les mots clefs faisant partie de la BDD dans la phrase saisie
 21 			var mot_cle = this.recherche_mot_cle(saisie_utilisateur);
 22 			//Si les nouveaux mots cles sont identiques à ceux de la demande précédente 
 23 			if ((saisie_utilisateur == this.derniers_messages[this.derniers_messages.length-2]) && (this.derniers_messages.length!=0))
 24 			{
 25 				mot_cle = ["sameinput"];
 26 			}
 27 			
 28 			//Traitement des mots reçus pour n'avoir que les mots cles les plus long ex : "graphe/ graphe connexe"
 29 			for (i in mot_cle)
 30 			{
 31 				for (j in mot_cle)
 32 				{
 33 					if ((String(mot_cle[j]).search(mot_cle[i]) != -1) && mot_cle[i] != mot_cle[j])
 34 					{
 35 						mot_cle.splice(i,1);
 36 					}
 37 				}
 38 			}
 39 
 40 			//Garde en mémoire si l'utilisateur n'a pas compris
 41 			var incomprehension = false;
 42 			//On garde la trace d'où est le mot cle qui signifie que l'utilisateur n'a pas compris ou qu'il veut un exemple
 43 			var trace;
 44 			//On retient en mémoire si l'utilisateur veut un exemple
 45 			var exemple = false;
 46 
 47 			for (i in mot_cle)
 48 			{
 49 				var correspondance = this.rechercher_correspondance(mot_cle[i]);
 50 				if (String(correspondance.ensemble_def)=="incomprehension")
 51 				{
 52 					incomprehension = true;
 53 					trace=i;
 54 				}
 55 				if (String(correspondance.ensemble_def)=="exemple")
 56 				{
 57 					exemple = true;
 58 					trace=i;
 59 				}
 60 
 61 			}
 62 
 63 			//S'il y a plus d'un mot clé, on demandera à l'utilisateur de quel notion il veut parler
 64 			if ((mot_cle.length>1) && (incomprehension==false) && (exemple == false))
 65 			{
 66 				this.choisir_mot_cle(mot_cle);
 67 			}
 68 			else //Sinon on trouve la réponse pour lui envoyer
 69 			{
 70 				var affichage; //Le message qui sera affiché au final
 71 				//Si on a repéré que l'utilisateur n'a pas compris
 72 				if (incomprehension == true)
 73 				{
 74 					if (mot_cle.length>1)
 75 					{
 76 						mot_cle.splice(trace, 1);
 77 						affichage = this.questionner_user(mot_cle);
 78 					}
 79 					else
 80 					{
 81 						affichage = this.questionner_user(this.derniers_mots_cles[this.derniers_mots_cles.length-1]);
 82 					}
 83 					this.afficher_reponse(affichage); //On affiche la question
 84 				}
 85 				//Si on a repéré que l'utilisateur veut un exemple
 86 				else if (exemple == true)
 87 				{
 88 					//s'il demande un exemple, on lui donne un exemple du dernier mot clef utilisé
 89 					if (mot_cle.length>1)
 90 					{
 91 						mot_cle.splice(trace, 1);
 92 						affichage = this.donner_exemple(mot_cle);
 93 					}
 94 					else
 95 					{
 96 						affichage = this.donner_exemple(this.derniers_mots_cles[this.derniers_mots_cles.length-1]);
 97 					}
 98 					this.afficher_reponse(affichage); //on affiche l'exemple
 99 				}
100 				else 
101 				{
102 					var reponse = this.rechercher_correspondance(mot_cle);
103 					if (typeof reponse!="undefined")
104 					{
105 						if (mot_cle != "sameinput")
106 						{
107 
108 							this.derniers_mots_cles.push(mot_cle);
109 						}
110 						affichage = reponse.ensemble_def; //On récupère la définition de la réponse
111 						this.afficher_reponse(affichage); //on affiche la définition
112 					}
113 					else
114 					{
115 						this.pasTrouve();
116 					}
117 				}
118 			}
119 			
120 		}
121 	}
122 	// Si (saisie_utilisateur[0] == "/")
123 	else
124 	{
125 		if (saisie_utilisateur.endsWith("admin")) //Afficher l'interface administrateur
126 		{
127 			admin(true);
128 			this.afficher("# INFO # L'interface administrateur a bien été dévoilée");
129 
130 		}
131 		else if (saisie_utilisateur.endsWith("adminoff")) //Cacher l'interface administrateur
132 		{
133 			admin(false);
134 			this.afficher("# INFO # L'interface administrateur à été cachée");
135 		}
136 		else if (saisie_utilisateur.indexOf("setname")!= -1) //Changer de pseudo
137 		{
138 			var pseudo = saisie_utilisateur.substr(9).trim() //On prendre le pseudo à partir du 9eme caractère
139 
140 			if(pseudo.length < 1) //Si l'utilisateur n'a pas renseigné de pseudo
141 	        {
142 	            this.afficher("# INFO # Vous devez renseigner votre pseudo");
143 	        }
144 	        else if(pseudo.length > 20) //Si l'utilisateur a mit un pseudo de plus de 20 caractères
145 	        {
146 	        	this.afficher("# INFO # Veuillez choisir un pseudo de moins de 20 caractères");
147 	        }
148 	        else //Si le pseudo respecte les conditions
149 	        {
150 	            this.afficher("# INFO # Votre nom a bien été modifié");
151 				modifier_nom(pseudo);
152 				console.log(pseudo);
153 	        }	
154 		}
155 		else
156 		{
157 			this.afficher("# INFO # La commande n'a paxs été reconnue");
158 		}
159 		document.getElementById("user_input_text").value = "";
160 	}
161 }
162 /**
163  * Fonction qui va demander à l'utilisateur de quel mot il veut parler
164  * 
165  * @param {Array} mot_cle { Le tableau des mots cles trouvés }
166  */
167 ElizaBot.prototype.choisir_mot_cle = function(mot_cle)
168 {
169  	var affichage = "Je ne comprend pas de quoi vous vouliez parler, voulez vous parler de ";
170 	//On parcourt tous les mot clef
171 	for (i in mot_cle)
172 	{
173 		//Si ce n'est pas le dernier mot
174 		if (i<mot_cle.length-1)
175 		{
176 			affichage += mot_cle[i]+", ou de ";
177 		}
178 		else if (i==mot_cle.length-1)
179 		{
180 			affichage += mot_cle[i]+"?";
181 		}
182 	}
183 	this.afficher_reponse(affichage);
184 }

186 /**
187  * Est appelée si la fonction donner_reponse n'a pas trouvé de réponse
188  *
189  * @method     pasTrouve
190  */
191 ElizaBot.prototype.pasTrouve = function()
192 {
193 	// 0 < proba < 1
194 	proba = Math.random();
195 	//Choisi au hasard une réponse
196 	if (proba<0.25)
197 	{
198 		this.afficher_reponse("Je ne comprend pas");
199 	}
200 	else if (proba<0.5)
201 	{
202 		this.afficher_reponse("Vous pouvez détailler?");
203 	}
204 	else if (proba<0.75)
205 	{
206 		this.afficher_reponse("Pardon?");
207 	}
208 	else
209 	{
210 		this.afficher_reponse("Dis m'en plus");
211 	}
212 }
