 65 /**
 66  * Fonction permettant de rechercher une reponse grace a des mots cles
 67  *
 68  * @method     rechercher_correspondance
 69  * @param      {Array}    mots_cle  { Le ou les mots clé à utiliser }
 70  * @return     {Reponse}  {Une reponse correspondante aux mots clé }
 71  */
 72 ElizaBot.prototype.rechercher_correspondance = function(mots_cle)
 73 {
 74 	for(var i in this.ensemble_rep)
 75 	{
 76 		for(var j in this.ensemble_rep[i].mots_cle)
 77 		{
 78 			if(mots_cle == this.ensemble_rep[i].mots_cle[j])
 79 			{
 80 				return this.ensemble_rep[i];
 81 			}
 82 		}
 83 	}
 84 };

331  * Permet de modifier le nom de l'utilisateur à tous moment dans le pogramme
332  *
333  * @method     modifier_nom
334  * @param      {(Function|string)}  value   { parameter_description }
335  */
336 function modifier_nom(value)
337 {
338 	if(typeof value == "undefined")
339 	{
340 		var okLong = false; //Déclaration variable booléen
341 	    
342 		while(okLong === false) //Boucle tant que la variable est fausse
343 	    {
344 	        nomUser = prompt("Quel est votre nom ?"); //Demande à l'utilisateur son prénom
345 	        if (nomUser == null)
346 	        {
347 	        	nomUser="user";
348 	        }
349 	        if(nomUser.length < 1) //Si le prénom est vide
350 	        {
351 	            alert("Veuillez ne pas rentrez un prénom vide !"); //Message d'alerte
352 	            okLong = false; //Variable tjr à faux
353 	        }
354 	        else if(nomUser.length > 20) //Si le prénom est trop long
355 	        {
356 	            alert("Vous rentrez un prénom trop long !"); //Message d'alerte
357 	            okLong = false; //Variable tjr à faux
358 	        }
359 	        else //C'est ok
360 	        {
361 	            okLong = true //On met la variable à l'état vrai
362 	        }
363 	        nomUser = nomUser.toUpperCase(); //On met le prénom en majuscule
364 	    }
365 	}
366 	else
367 	{
368 		nomUser = value;
369 	}
370 
371 	//Reinitialisation des espaces
372 	espace= "" ;
373 	espace1= "" ;
374 	
375 	createCookie("usernameElizaBot", nomUser, 7);
376     if((nomUser.length) > (5)) //Si longueur nom supérieur à 5
377     {
378         nbespace = (nomUser.length) - (5); //On calcule différence
379         for(var z = 0; z < nbespace; z++) //Boucle de longueur de la différence

386         nbespace1 = 5 - (nomUser.length); //On calcule différence
387         for(var y = 0; y < nbespace1; y++) //Boucle de longueur de la différence
388         {
389             espace1 += " "; //On ajoute des espaces
390         }
391     }
