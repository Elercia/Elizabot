1 /**
  2  * Méthode permettant de poser des questions à l'utilisateur 
  3  * en gérant la dépendance des connaissances
  4  *
  5  * @method     questionner_user
  6  * @param      {String}  mot_clef  L'ensemble des mots clés correspondant à la question à poser 
  7  * @return     {string}  Le string correspondant à la question à poser 
  8  */
  9 ElizaBot.prototype.questionner_user = function(mot_clef)
 10 {
 11 	var resultat = "Quelle notion ne comprend tu pas : Celle de ";
 12 
 13 	var rep1 = this.rechercher_correspondance(mot_clef);//on recherche la reponse correspondant au mot clé que l'on nous a donné
 14 
 15 	if(!(rep1 instanceof Reponse))
 16 	{
 17 		resultat = "Je ne peux malheureusement pas t'aider";
 18 	}
 19 	else if (rep1.ensemble_dependance=[]) 
 20 	{
 21 		resultat = "Je ne peut pas plus développer que ça";
 22 	}
 23 	else
 24 	{
 25 		var dep = rep1.ensemble_dependance;//on récupère ces dépendances
 26 		
 27 		if(Array.isArray(dep))
 28 			resultat += dep.join(" ou de ");
 29 		else
 30 			resultat += dep;
 31 	}
 32 	resultat += "?";
 33 
 34 	return resultat;  
 35 }
